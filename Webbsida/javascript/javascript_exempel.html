Byt till slumpad bakgrundsfärg

Denna koden skall ligga i din javascript fil

function randomColor() {
          var letters = "0123456789ABCDEF";
          var color = "#";
          for (var i = 0; i < 6; i++) {
            color += letters[Math.floor(Math.random() * 16)];
          }
          document.body.style.backgroundColor = color;
        }

Denna koden skall ligga i din html fil

<button onclick="randomColor()">Byt färg</button>

var bildIndex = 1;
          function bytBild() {
            var bild = document.getElementById("minBild");
            bildIndex++;
            if (bildIndex > 5) {
              bildIndex = 1;
            }
            bild.src = "bilder/bild" + bildIndex + ".jpg";
          }

Denna koden skall ligga i din html fil

<button onclick="bytBild()">Byt Bild</button>
